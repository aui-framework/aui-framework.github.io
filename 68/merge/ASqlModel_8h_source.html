89 IncompleteSelectRequest& where(const ASqlBuilder::WhereStatement::WhereExpr& w) {
90 auto[exprString, whereParams] = ASqlBuilder::WhereStatement::WhereExpr::unpack(w);
91 if (!exprString.empty()) {
92 if (mWhereExpr.empty()) {
93 mWhereExpr = "WHERE " + exprString;
94 mWhereParams = std::move(whereParams);
95 } else {
96 mWhereExpr += "AND " + exprString;
97 mWhereParams.insertAll(whereParams);
98 }
99 }
100 return *this;
101 }

107 AVector<Model> get() {
108 auto idField = AField<ASqlModel<Model>>::make(&ASqlModel<Model>::id);
109 AVector<Model> result;
110 result.reserve(0x100);
111
112 mSql += " ";
113 mSql += mWhereExpr;
114 auto dbResult = Autumn::get<ASqlDatabase>()->query(mSql, mWhereParams);
115
116 AVector<size_t> sqlColumnToModelFieldIndexMapping;
117 AVector<_<AField<Model>>> fields;
118 fields << AModelMeta<Model>::getFields().valueVector();
119
120 for (auto& row : dbResult) {
121 Model m;
122 idField->set(m, row->getValue(0));
123 for (size_t columnIndex = 1; columnIndex < dbResult->getColumns().size(); ++columnIndex) {
124 fields[columnIndex - 1]->set(m, row->getValue(columnIndex));
125 }
126 result << std::move(m);
127 }
128
129 return result;
130 }

138 inline Model first() {
139 auto result = get();
140 if (result.size() == 0)
141 throw NoSuchRowException();
142 if (result.size() != 1)
143 throw TooManyRowsException();
144 return result.first();
145 }

161 static Model byId(id_t id) {
162 AStringVector columns;
163 columns << "id";
164 columns << Meta::getFields().keyVector();
165 auto result = table(Meta::getSqlTable()).select(columns).where(col("id") == id)
166 .template as<Model>();
167
168 if (result.empty())
169 throw NoSuchRowException();
170 assert(result.size() == 1);
171 return result.first();
