65 if constexpr (std::is_same_v<void, T>) {
66 auto impl = future.inner().get();
67 mFutureSet << std::move(future);
68 lock.unlock();

97 void waitForAll() {
98 std::unique_lock lock(mSync);
99 if (!mFutureSet.empty()) {
100 waitAgain:
101 auto futureSet = std::move(mFutureSet);
102 lock.unlock();
103 futureSet.waitForAll();
104 lock.lock();
105 if (mFutureSet.empty()) {
106 mFutureSet = std::move(futureSet);
107 } else {
108 mFutureSet.insertAll(futureSet);
109 goto waitAgain;
110 }
111 }

113 while (!mCustomTypeFutures.empty()) {
114 mCustomTypeFutures.front()->wait(lock);
