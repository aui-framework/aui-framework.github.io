198 _<AOverlappingSurface> createOverlappingSurface(const glm::ivec2& position,
199 const glm::ivec2& size,
200 bool closeOnClick = true) {
201 return createOverlappingSurface([&](unsigned attempt) -> AOptional<glm::ivec2> {
202 switch (attempt) {
203 case 0: return position;
204 case 1: return glm::clamp(position, {0, 0}, {getSize() - size});
205 default: return std::nullopt;
206 }
207 }, size, closeOnClick);
208 }
209
220 _<AOverlappingSurface> createOverlappingSurface(const std::function<AOptional<glm::ivec2>(unsigned)>& positionFactory,
221 const glm::ivec2& size,
222 bool closeOnClick = true) {
223 glm::ivec2 position = {0, 0};
224 auto maxPos = getSize() - size;
225 for (unsigned index = 0; ; ++index) {
226 auto optionalPosition = positionFactory(index);
227 if (optionalPosition) {
228 position = *optionalPosition;
229
230 if (position.x >= 0 && position.y >= 0 && glm::all(glm::lessThan(position, maxPos))) {
231 break;
232 }
233 } else {
234 break;
235 }
236 }
237
238 auto tmp = createOverlappingSurfaceImpl(position, size);
239 tmp->mParentWindow = this;
240 tmp->mCloseOnClick = closeOnClick;
241 mOverlappingSurfaces << tmp;
242 return tmp;
243 }
244 void closeOverlappingSurface(AOverlappingSurface* surface) {
245 if (mOverlappingSurfaces.erase(aui::ptr::fake(surface)) > 0) {
246 closeOverlappingSurfaceImpl(surface);
247 }
248 }
