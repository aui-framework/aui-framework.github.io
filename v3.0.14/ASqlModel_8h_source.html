84 IncompleteSelectRequest& where(const ASqlBuilder::WhereStatement::WhereExpr& w) {
85 auto[exprString, whereParams] = ASqlBuilder::WhereStatement::WhereExpr::unpack(w);
86 if (!exprString.empty()) {
87 if (mWhereExpr.empty()) {
88 mWhereExpr = "WHERE " + exprString;
89 mWhereParams = std::move(whereParams);
90 } else {
91 mWhereExpr += "AND " + exprString;
92 mWhereParams.insertAll(whereParams);
93 }
94 }

102 AVector<Model> get() {
103 auto idField = AField<ASqlModel<Model>>::make(&ASqlModel<Model>::id);
104 AVector<Model> result;
105 result.reserve(0x100);
106
107 mSql += " ";
108 mSql += mWhereExpr;
109 auto dbResult = Autumn::get<ASqlDatabase>()->query(mSql, mWhereParams);
110
111 AVector<size_t> sqlColumnToModelFieldIndexMapping;
112 AVector<_<AField<Model>>> fields;
113 fields << AModelMeta<Model>::getFields().valueVector();
114
115 for (auto& row : dbResult) {
116 Model m;
117 idField->set(m, row->getValue(0));
118 for (size_t columnIndex = 1; columnIndex < dbResult->getColumns().size(); ++columnIndex) {
119 fields[columnIndex - 1]->set(m, row->getValue(columnIndex));
120 }
121 result << std::move(m);
122 }
123
124 return result;
125 }
