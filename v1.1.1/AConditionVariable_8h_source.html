37 struct WaitHelper {
38 WaitHelper(AConditionVariable& var) noexcept {
39 auto thread = AThread::current();
40 std::unique_lock lock(thread->mCurrentCV.mutex);
41 thread->mCurrentCV.cv = &var;
42 }
43 ~WaitHelper() noexcept(false) {
44 auto thread = AThread::current();
45 std::unique_lock lock(thread->mCurrentCV.mutex);
46 thread->mCurrentCV.cv = nullptr;
47 }
48 };

50 template<typename Predicate>
51 struct PredicateHelper {
52 Predicate predicate;
53 _<AAbstractThread> thread;
54
55 PredicateHelper(Predicate predicate) : predicate(std::move(predicate)), thread(AThread::current()) {}
56 bool operator()() noexcept {
57 if (thread->isInterrupted()) return true;
58 return predicate();
59 }
60 };
