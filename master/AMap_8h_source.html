183 template<typename Factory>
184 ValueType& getOrInsert(const KeyType& keyType, Factory&& factory) noexcept(noexcept(factory())) {
185 auto[it, isElementCreated] = parent::insert(typename parent::value_type(keyType, ValueType{}));
186 static_assert(std::is_same_v<decltype(it), typename parent::iterator>, "govno");
187 if (isElementCreated) {

193 template<typename BinaryOperation>
194 auto toVector(BinaryOperation&& transformer) const -> AVector<decltype(transformer(std::declval<KeyType>(), std::declval<ValueType>()))> {
195 AVector<decltype(transformer(std::declval<KeyType>(), std::declval<ValueType>()))> result;
196 result.reserve(parent::size());
197 std::transform(parent::begin(), parent::end(), std::back_inserter(result), [transformer = std::forward<BinaryOperation>(transformer)](const typename parent::value_type& p){
198 return transformer(p.first, p.second);
199 });
200 return result;

205template<typename Iterator, typename UnaryOperation>
206inline auto aui::container::to_map(Iterator begin,
207 Iterator end,
208 UnaryOperation&& transformer) {
209 AMap<decltype(transformer(*begin).first),
210 decltype(transformer(*begin).second)> result;
211
212 for (auto it = begin; it != end; ++it) {
213 auto[key, value] = transformer(*it);
214 result[std::move(key)] = std::move(value);
215 }
216 return result;
