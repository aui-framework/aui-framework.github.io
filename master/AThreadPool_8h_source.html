133 inline auto operator*(Callable fun)

135 using Value = std::invoke_result_t<Callable>;
136 AFuture<Value> future(std::move(fun));
137 run([innerWeak = future.inner()->wrapped.weak()]()

139 auto innerUnsafePointer = innerWeak.lock().get(); // using .get() here in order to bypass null check in

142 }, AThreadPool::PRIORITY_LOWEST);

170template<typename Iterator, typename Functor>
171auto AThreadPool::parallel(Iterator begin, Iterator end, Functor&& functor) {
172 using ResultType = decltype(std::declval<Functor>()(std::declval<Iterator>(), std::declval<Iterator>()));
173 AFutureSet<ResultType> futureSet;
174
175 size_t itemCount = end - begin;
176 size_t affinity = (glm::min)(AThreadPool::global().getTotalWorkerCount(), itemCount);
177 if (affinity == 0) return futureSet;
178 size_t itemsPerThread = itemCount / affinity;
179
180 for (size_t threadIndex = 0; threadIndex < affinity; ++threadIndex) {
181 auto forThreadBegin = begin;
182 begin += itemsPerThread;
183 auto forThreadEnd = threadIndex + 1 == affinity ? end : begin;
184 futureSet.push_back(*this * [functor = std::forward<Functor>(functor), forThreadBegin, forThreadEnd]() -> decltype(auto) {
185 return functor(forThreadBegin, forThreadEnd);
186 });
187 }
188
189 return futureSet;
190}
