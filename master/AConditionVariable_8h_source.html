26class AConditionVariable

28private:

31 struct WaitHelper {
32 WaitHelper(AConditionVariable& var) noexcept {
33 auto thread = AThread::current();
34 std::unique_lock lock(thread->mCurrentCV.mutex);
35 thread->mCurrentCV.cv = &var;
36 }
37 ~WaitHelper() noexcept(false) {
38 auto thread = AThread::current();
39 std::unique_lock lock(thread->mCurrentCV.mutex);
40 thread->mCurrentCV.cv = nullptr;
41 }
42 };

44 template<typename Predicate>
45 struct PredicateHelper {
46 Predicate predicate;
47 _<AAbstractThread> thread;
48
49 PredicateHelper(Predicate predicate) : predicate(std::move(predicate)), thread(AThread::current()) {}
50 bool operator()() noexcept {
51 if (thread->isInterrupted()) return true;
52 return predicate();
53 }
54 };

56public:

73 void notify_one() noexcept { mImpl.notify_one(); }

75 template<typename Lock>
