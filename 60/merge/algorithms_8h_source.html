54 template<typename Iterator, typename Predicate>
55 Iterator binary_search(Iterator begin, Iterator end, Predicate predicate) {
56 size_t rangeEnd = end - begin;
57 size_t rangeBegin = 0;
58
59
60 for (size_t i; rangeBegin < rangeEnd; ) {
61 i = (rangeEnd - rangeBegin - 1) / 2 + rangeBegin;
62 auto it = begin + i;
63 switch (predicate(it)) {
64 case 0:
65 return it;
66
67 case 1:
68 rangeBegin = i + 1;
69 break;
70 case -1:
71 rangeEnd = i;
72 break;
73 }
74
75 }
