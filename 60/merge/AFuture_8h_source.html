101 struct Inner {
102 bool interrupted = false;
106 std::conditional_t<isVoid, bool, std::optional<Value>> value;
107 std::optional<AInvocationTargetException> exception;
108 AMutex mutex;
109 AConditionVariable cv;
110 TaskCallback task;
111 OnSuccessCallback onSuccess;
112 std::function<void(const AException& exception)> onError;
113 _<AAbstractThread> thread;
114 bool cancelled = false;
115
116 explicit Inner(std::function<Value()> task) noexcept: task(std::move(task)) {
117 if constexpr(isVoid) {
118 value = false;
119 }
120 }
121
122 void waitForTask() noexcept {
123 std::unique_lock lock(mutex);
124 bool rethrowInterrupted = false;
125 while ((thread) && !hasResult()) {
126 try {
127 cv.wait(lock);
128 } catch (const AThread::Interrupted& i) {
129 rethrowInterrupted = true;
130 }
131 }
132 if (rethrowInterrupted) {
133 AThread::current()->interrupt();
134 }
135 }
136
137 [[nodiscard]]
138 bool isWaitNeeded() noexcept {
139 return (thread || !cancelled) && !hasResult();
140 }
141
142 [[nodiscard]]
143 bool hasResult() const noexcept {
144 return value || exception || interrupted;
145 }
146
147 [[nodiscard]]
148 bool hasValue() const noexcept {
149 return bool(value);
150 }
151
152 bool setThread(_<AAbstractThread> thr) noexcept {
153 std::unique_lock lock(mutex);
154 if (cancelled) return true;
155 if (thread) return true;
156 thread = std::move(thr);
157 return false;
158 }
159
160 void wait(const _weak<Inner>& innerWeak, AFutureWait flags = AFutureWait::DEFAULT) noexcept {
161 std::unique_lock lock(mutex);
162 try {
163 if ((thread || !cancelled) && !hasResult() && int(flags) & 0b1 && task) {
164 // task is not have picked up by the threadpool; execute it here
165 lock.unlock();
166 if (tryExecute(innerWeak)) {
167 return;
168 }
169 lock.lock();
170 }
171 while ((thread || !cancelled) && !hasResult()) {
172 cv.wait(lock);
173 }
174 } catch (const AThread::Interrupted& e) {
175 e.needRethrow();
176 }
177 }
178
179 void cancel() noexcept {
180 std::unique_lock lock(mutex);
181 if (!cancelled) {
182 cancelled = true;
183 if (thread && !hasResult()) {
184 thread->interrupt();
185 }
186 }
187 }
188
195 bool tryExecute(const _weak<Inner>& innerWeak) {
196 /*
197 * We should assume that this == nullptr or invalid.
198 * We can assert that this pointer is safe only if we hold at least one shared_ptr.
199 * It allows callers to pass a weak_ptr in any state.
200 */
201 if (auto inner = innerWeak.lock()) {
202 if (value) return false;
203 if (inner->setThread(AThread::current())) return false;
204 try {
205 if (task == nullptr) { // task is executed in wait() function
206 return false;
207 }
208 std::unique_lock lock(mutex);
209 if (task == nullptr) { // task is executed in wait() function
210 return false;
211 }
212 auto func = std::move(task);
213 assert(bool(func));
214 lock.unlock();
215 inner = nullptr;
216 if constexpr(isVoid) {
217 func();
218 if (auto sharedPtrLock = innerWeak.lock()) {
219 std::unique_lock lock(mutex);
220 value = true;
221 cv.notify_all();
222 nullsafe(onSuccess)();
223 }
224 } else {
225 auto result = func();
226 if (auto sharedPtrLock = innerWeak.lock()) {
227 std::unique_lock lock(mutex);
228 value = std::move(result);
229 cv.notify_all();
230 nullsafe(onSuccess)(*value);
231 }
232 }
233 } catch (const AException& e) {
234 nullsafe(innerWeak.lock())->reportException();
235 return false;
236 } catch (...) {
237 nullsafe(innerWeak.lock())->reportInterrupted();
238 throw;
239 }
240 }
241 return true;
242 }
243
244 void reportInterrupted() noexcept {
245 std::unique_lock lock(mutex);
246 interrupted = true;
247 cv.notify_all();
248 }
249
250 void reportException() noexcept {
251 std::unique_lock lock(mutex);
252 exception.emplace();
253 cv.notify_all();
254 nullsafe(onError)(*exception);
255 }
256
257 };

303 template<typename Callback>
304 Future& onSuccess(Callback&& callback) noexcept {
305 std::unique_lock lock((*mInner)->mutex);
306 if constexpr(isVoid) {
307 if ((*mInner)->onSuccess) {
308 (*mInner)->onSuccess = [prev = std::move((*mInner)->onSuccess),
309 innerStorage = mInner,
310 callback = std::forward<Callback>(callback)]() {
311 prev();
312 callback();
313 };
314 } else {
315 (*mInner)->onSuccess = [innerStorage = mInner, callback = std::forward<Callback>(callback)]() {
316 callback();
317 };
318 }
319 } else {
320 if ((*mInner)->onSuccess) {
321 (*mInner)->onSuccess = [prev = std::move((*mInner)->onSuccess),
322 innerStorage = mInner,
323 callback = std::forward<Callback>(callback)](const Value& v) {
324 prev(v);
325 callback(v);
326 };
327 } else {
328 (*mInner)->onSuccess = [innerStorage = mInner, callback = std::forward<Callback>(callback)](
329 const Value& v) {
330 callback(v);
331 };
332 }
333 }
334 return *this;
335 }

339 std::unique_lock lock((*mInner)->mutex);
340
341 if ((*mInner)->onError) {
342 (*mInner)->onError = [prev = std::move((*mInner)->onError),
343 innerStorage = mInner,
344 callback = std::forward<Callback>(callback)](const AException& v) {
345 prev(v);
346 callback(v);
347 };
348 } else {
349 (*mInner)->onError = [innerStorage = mInner, callback = std::forward<Callback>(callback)](
350 const AException& v) {
351 callback(v);
352 };
353 }
354 return *this;
355 }

383 decltype(auto) get(AFutureWait flags = AFutureWait::DEFAULT) {
384 AThread::interruptionPoint();
385 (*mInner)->wait(mInner->sharedPtr(), flags);
386 AThread::interruptionPoint();
387 if ((*mInner)->exception) {
388 throw *(*mInner)->exception;
389 }

392 }
393 if constexpr(!isVoid) {
394 return *(*mInner)->value;
395 }

451template<typename T = void>
452class AFuture final: public aui::impl::future::Future<T> {
453private:
454 using super = typename aui::impl::future::Future<T>;
455
456public:
457 using Task = typename super::TaskCallback;
458
459 AFuture(Task task = nullptr) noexcept: super(std::move(task)) {}
460
461 void supplyResult(T v) noexcept {
462 auto& inner = (*super::mInner);
463 assert(("task is already provided", inner->task == nullptr));
464
465 std::unique_lock lock(inner->mutex);
466 inner->value = std::move(v);
467 inner->cv.notify_all();
468 nullsafe(inner->onSuccess)(*inner->value);
469 }
470
471 AFuture& operator=(std::nullptr_t) noexcept {
472 super::mInner = nullptr;
473 return *this;
474 }
475
476 [[nodiscard]]
477 bool operator==(const AFuture& r) const noexcept {
478 return super::mInner == r.mInner;
479 }
480};
