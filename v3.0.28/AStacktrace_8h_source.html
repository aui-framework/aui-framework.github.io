144inline std::ostream& operator<<(std::ostream& o, const AStacktrace& stacktrace) noexcept {
145 stacktrace.resolveSymbolsIfNeeded();
146 for (const auto& entry : stacktrace) {
147 o << " - at " << entry.ptr();
148 if (auto name = entry.functionName()) {
149 o << " " << *name;
150 }
151 o << "(";
152
153 if (auto filename = entry.fileName()) {
154 o << *filename;
155 } else {
156 o << "?";
157 }
158 o << ":";
159 if (auto line = entry.lineNumber()) {
160 o << *line;
161 } else {
162 o << "?";
163 }
164
165 o << ")" << std::endl;
166 }
167 return o;
168}
