165 template<typename OtherIterator>
166 constexpr iterator insert(iterator at, OtherIterator begin, OtherIterator end) {
167 AUI_ASSERT_MY_ITERATOR(at);
168 auto distance = std::distance(begin, end);
169
170 if (size() + distance <= reserved()) {
171 return aui::container::vector_impl::insert_no_growth(mEnd, at, begin, end);
172 }
173 ADynamicVector temp;
174 temp.reserve(aui::container::vector_impl::ceilPower2(distance + size()));
175 aui::container::vector_impl::insert_no_growth(temp.mEnd, temp.mEnd,
176 std::make_move_iterator(mBegin), std::make_move_iterator(at));
177
178 auto result = aui::container::vector_impl::insert_no_growth(temp.mEnd, temp.mEnd,
179 begin, end);
180
181 aui::container::vector_impl::insert_no_growth(temp.mEnd, temp.mEnd,
182 std::make_move_iterator(at), std::make_move_iterator(mEnd));
183 operator=(std::move(temp));
184
185 return result;
186
187 }

205 void reserve(std::size_t newSize) {
206 if (reserved() == newSize) {
207 return;
208 }
209
210 auto newBuffer = static_cast<StoredType*>(operator new[](newSize * sizeof(StoredType)));
211 auto newBufferEnd = newBuffer;
212 try {
213 auto elementsToMove = std::min(newSize, size());
214 auto moveFrom = begin();
215 for (std::size_t i = 0; i < elementsToMove; ++newBufferEnd, ++moveFrom) {
216 new (newBufferEnd) StoredType(std::move(*moveFrom));
217 }
218 deallocate();
219 mBegin = newBuffer;
220 mEnd = newBufferEnd;
221 mBufferEnd = newBuffer + newSize;
222 } catch (...) { // unlikely; but just in case
223 delete[] newBuffer;
224 throw;
225 }
226 }
