172 template<typename OtherIterator>
173 constexpr iterator insert(iterator at, OtherIterator begin, OtherIterator end) {
174 AUI_ASSERT_MY_ITERATOR(at);
175 auto distance = std::distance(begin, end);
176
177 if (distance + size() <= StaticVectorSize) {
178 return inplace()->insert(at, begin, end);
179 }
180 if (!isInplaceAllocated()) {
181 return dynamic()->insert(at, begin, end);
182 }
183
184 // switch from static to dynamic
185 DynamicVector temp;
186 temp.reserve(aui::container::vector_impl::ceilPower2(distance + size()));
187
188 aui::container::vector_impl::insert_no_growth(temp.mEnd, temp.mEnd,
189 std::make_move_iterator(this->begin()), std::make_move_iterator(at));
190
191 auto result = aui::container::vector_impl::insert_no_growth(temp.mEnd, temp.mEnd,
192 begin, end);
193
194 aui::container::vector_impl::insert_no_growth(temp.mEnd, temp.mEnd,
195 std::make_move_iterator(at), std::make_move_iterator(this->end()));
196
197 inplace()->~StaticVector();
198 new (&mBase.dynamic) DynamicVector(std::move(temp));
199
200 return result;
201 }

203 constexpr iterator erase(iterator begin, iterator end) noexcept {
204 AUI_ASSERT_MY_ITERATOR(begin);
205 AUI_ASSERT_MY_ITERATOR(end);
206
207 if (isInplaceAllocated()) {
208 return inplace()->erase(begin, end);
209 }
210
211 if (size() - std::distance(begin, end) > StaticVectorSize) {
212 return dynamic()->erase(begin, end);
213 }
214
215 auto index = std::distance(this->begin(), begin);
216
217 // switch from dynamic to static
218 auto temp = std::move(*dynamic());
219 temp.erase(begin, end);
220 new (inplace()) StaticVector();
221 inplace()->insert(inplace()->end(), std::make_move_iterator(temp.begin()), std::make_move_iterator(temp.end()));
222 return this->begin() + index;
223 }
