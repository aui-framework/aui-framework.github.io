87#define AUI_JNI_STATIC_METHOD(ret_t, name, args) \
88 static ret_t name (AUI_PP_FOR_EACH(AUI_JNI_INTERNAL_OMIT_BRACES, _, args)) { \
89 static_assert(::aui::jni::convertible<ret_t>, "return type is required to be convertible"); \
90 auto clazz = getClass(); \
91 auto e = ::aui::jni::env(); \
92 const char* signature = ::aui::jni::signature_v<ret_t (AUI_PP_FOR_EACH(AUI_JNI_INTERNAL_OMIT_BRACES, _, args))>; \
93 static auto methodId = e->GetStaticMethodID(clazz, #name, signature); \
94 if (methodId == 0) { \
95 throw AException("no such static jni method: {} {}"_format(#name, signature)); \
96 } \
97 return ::aui::jni::callStaticMethod<ret_t>(clazz, methodId AUI_PP_FOR_EACH(AUI_JNI_INTERNAL_OMIT_BRACES_CONTENTS, _, args)); \
98 }

127 static auto methodId = e->GetMethodID(clazz, #name, signature); \
