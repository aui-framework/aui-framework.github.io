99 char readChar() {
100 if (mReverse) {
101 mReverse = false;
102 return mLastByte;
103 }
104
105 if (mBufferRead >= mBufferEnd) {
106 // read next blob
107 mBufferEnd = mBuffer + mInput->read(mBuffer, sizeof(mBuffer));
108 mBufferRead = mBuffer;
109 if (mBufferEnd == mBufferRead) {
110 throw AEOFException();
111 }
112 }
113
114 mLastByte = *(mBufferRead++);
115
116 if (mLastByte == '\n')
117 {
118 mRow += 1;
119 mColumn = 1;
120 } else
121 {
122 mColumn += 1;
123 }
124 return mLastByte;
125 }
