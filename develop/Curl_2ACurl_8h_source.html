209 Builder& withBody(std::string contents) {
210 assert(("write callback already set" && mReadCallback == nullptr));
211
212 struct Body {
213 explicit Body(std::string b) : contents(std::move(b)), i(contents.begin()) {}
214
215 std::string contents;
216 std::string::iterator i;
217 };
218 auto b = _new<Body>(std::move(contents));
219
220 mReadCallback = [body = std::move(b)](char* dst, std::size_t length) mutable {
221 if (body->i == body->contents.end()) {
222 throw AEOFException();
223 }
224 std::size_t remaining = std::distance(body->i, body->contents.end());
225 length = glm::min(length, remaining);
226 std::memcpy(dst, &*body->i, length);
227 body->i += length;
228 return length;
229 };
230
231 return *this;
232 }

240 mHeaderCallback = std::move(headerCallback);
