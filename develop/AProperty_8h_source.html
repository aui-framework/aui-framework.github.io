  227struct APropertyDef {
  231    const M* base;
  232    using Model = M;
  233
  237    Getter get;
  238
  254    Setter set;
  255    using GetterReturnT = decltype(std::invoke(get, base));
  256    using Underlying = std::decay_t<GetterReturnT>;
  257
  261    const emits<SignalArg>& changed;
  262    static_assert(aui::same_as<Underlying , std::decay_t<SignalArg>>, "different getter result and signal arg?");
  263
  264    // this ctor effectively prohibits designated initialization, i.e., this one is not possible:
  265    //
  266    // auto size() const {
  267    //     return APropertyDef {
  268    //         .base = this,
  269    //         .get = &AView::mSize,
  270    //         .set = &AView::setSize,
  271    //         .changed = mSizeChanged,
  272    //     };
  273    // }
  274    //
  275    // deduction in designated initializers is relatively recent feature.
  276    APropertyDef(const M* base, Getter get, Setter set, const emits<SignalArg>& changed)
  277      : base(base), get(std::move(get)), set(std::move(set)), changed(changed) {}

  280    APropertyDef& operator=(U&& u) {
  281        std::invoke(set, *const_cast<Model*>(base), std::forward<U>(u));
  282        return *this;
  283    }

  293    }
  294

  297        return &std::invoke(get, base);
  298    }
  299
