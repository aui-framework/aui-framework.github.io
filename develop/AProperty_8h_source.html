  220template <
  221    typename M, aui::invocable<M&> Getter, aui::invocable<M&, std::invoke_result_t<Getter, M&>> Setter,
  222    typename SignalArg>
  223struct APropertyDef {
  227    const M* base;
  228    using Model = M;
  229
  233    Getter get;
  234
  250    Setter set;
  251    using GetterReturnT = decltype(std::invoke(get, base));
  252    using Underlying = std::decay_t<GetterReturnT>;
  253
  257    const emits<SignalArg>& changed;
  258    static_assert(aui::same_as<Underlying , std::decay_t<SignalArg>>, "different getter result and signal arg?");
  259
  260    // this ctor effectively prohibits designated initialization, i.e., this one is not possible:
  261    //
  262    // auto size() const {
  263    //     return APropertyDef {
  264    //         .base = this,
  265    //         .get = &AView::mSize,
  266    //         .set = &AView::setSize,
  267    //         .changed = mSizeChanged,
  268    //     };
  269    // }
  270    //
  271    // deduction in designated initializers is relatively recent feature.
  272    APropertyDef(const M* base, Getter get, Setter set, const emits<SignalArg>& changed)
  273      : base(base), get(std::move(get)), set(std::move(set)), changed(changed) {}
  274
  275    template <aui::convertible_to<Underlying> U>
  276    APropertyDef& operator=(U&& u) {
  277        std::invoke(set, *const_cast<Model*>(base), std::forward<U>(u));
  278        return *this;
  279    }
  280
  281    [[nodiscard]]
  282    GetterReturnT value() const noexcept {
  283        return std::invoke(get, base);
  284    }
  285
  286    [[nodiscard]]
  287    GetterReturnT operator*() const noexcept {
  288        return std::invoke(get, base);
  289    }
  290
  291    [[nodiscard]]
  292    const Underlying* operator->() const noexcept {
  293        return &std::invoke(get, base);
  294    }
  295
  296    [[nodiscard]] operator GetterReturnT() const noexcept { return std::invoke(get, base); }
  297
  298    [[nodiscard]]
  299    M* boundObject() const {
  300        return const_cast<M*>(base);
  301    }

  307    [[nodiscard]]
