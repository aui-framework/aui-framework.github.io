355 inline static AVector<StoredType, Allocator> generate(size_t size, Callable&& callable) noexcept {
356 AVector<StoredType, Allocator> s;
357 s.reserve(size);
358 for (size_t i = 0; i < size; ++i) {
359 s << callable(i);
