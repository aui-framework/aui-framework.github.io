280 template<typename Callable>
281 inline static AVector<StoredType, Allocator> generate(size_t size, Callable&& callable) noexcept {
282 AVector<StoredType, Allocator> s;
283 s.reserve(size);
284 for (size_t i = 0; i < size; ++i) {
285 s << callable(i);
