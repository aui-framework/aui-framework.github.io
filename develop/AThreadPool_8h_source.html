117 [[nodiscard]]
118 inline auto operator*(Callable fun)
119 {
120 using Value = std::invoke_result_t<Callable>;
121 AFuture<Value> future(std::move(fun));
122 run([innerWeak = future.inner()->wrapped.weak()]()
123 {
124 auto innerUnsafePointer = innerWeak.lock().get(); // using .get() here in order to bypass null check in
125 // operator->
126 innerUnsafePointer->tryExecute(innerWeak);
127 }, AThreadPool::PRIORITY_LOWEST);
128 return future;
129 }

155template<typename Iterator, typename Functor>
156auto AThreadPool::parallel(Iterator begin, Iterator end, Functor&& functor) {
157 using ResultType = decltype(std::declval<Functor>()(std::declval<Iterator>(), std::declval<Iterator>()));
158 AFutureSet<ResultType> futureSet;
159
160 size_t itemCount = end - begin;
161 size_t affinity = (glm::min)(AThreadPool::global().getTotalWorkerCount(), itemCount);
162 if (affinity == 0) return futureSet;
163 size_t itemsPerThread = itemCount / affinity;
164
165 for (size_t threadIndex = 0; threadIndex < affinity; ++threadIndex) {
166 auto forThreadBegin = begin;
167 begin += itemsPerThread;
168 auto forThreadEnd = threadIndex + 1 == affinity ? end : begin;
169 futureSet.push_back(*this * [functor = std::forward<Functor>(functor), forThreadBegin, forThreadEnd]() -> decltype(auto) {
170 return functor(forThreadBegin, forThreadEnd);
171 });
172 }
173
174 return futureSet;
175}
