100 template<typename Value = void>
101 class Future
102 {
103 public:
104 static constexpr bool isVoid = std::is_same_v<void, Value>;
105 using TaskCallback = std::function<Value()>;
106
107 using OnSuccessCallback = typename OnSuccessCallback<Value>::type;
108
109 struct Inner {
110 bool interrupted = false;
114 std::conditional_t<isVoid, bool, AOptional<Value>> value;
115 AOptional<AInvocationTargetException> exception;
116 AMutex mutex;
117 AConditionVariable cv;
118 TaskCallback task;
119 OnSuccessCallback onSuccess;
120 std::function<void(const AException& exception)> onError;
121 _<AAbstractThread> thread;
122 bool cancelled = false;
123
124 explicit Inner(std::function<Value()> task) noexcept: task(std::move(task)) {
125 if constexpr(isVoid) {
126 value = false;
127 }
128 }
129
130 void waitForTask() noexcept {
131 std::unique_lock lock(mutex);
132 bool rethrowInterrupted = false;
133 while ((thread) && !hasResult() && !cancelled) {
134 try {
135 cv.wait(lock);
136 } catch (const AThread::Interrupted&) {
137 rethrowInterrupted = true;
138 }
139 }
140 if (rethrowInterrupted) {
141 AThread::current()->interrupt();
142 }
143 }
144
145 [[nodiscard]]
146 bool isWaitNeeded() noexcept {
147 return (thread || !cancelled) && !hasResult();
148 }
149
150 [[nodiscard]]
151 bool hasResult() const noexcept {
152 return value || exception || interrupted;
153 }
154
155 [[nodiscard]]
156 bool hasValue() const noexcept {
157 return bool(value);
158 }
159
160 bool setThread(_<AAbstractThread> thr) noexcept {
161 std::unique_lock lock(mutex);
162 if (cancelled) return true;
163 if (thread) return true;
164 thread = std::move(thr);
165 return false;
166 }
167
168 void wait(const _weak<CancellationWrapper<Inner>>& innerWeak, AFutureWait flags = AFutureWait::DEFAULT) noexcept {
169 std::unique_lock lock(mutex);
170 try {
171 if ((thread || !cancelled) && !hasResult() && int(flags) & int(AFutureWait::DEFAULT) && task) {
172 // task is not have picked up by the threadpool; execute it here
173 lock.unlock();
174 if (tryExecute(innerWeak)) {
175 return;
176 }
177 lock.lock();
178 }
179 while ((thread || !cancelled) && !hasResult()) {
180 cv.wait(lock);
181 }
182 } catch (const AThread::Interrupted& e) {
183 e.needRethrow();
184 }
185 }
186
187 void cancel() noexcept {
188 std::unique_lock lock(mutex);
189 if (!cancelled) {
190 cancelled = true;
191 if (thread && !hasResult()) {
192 thread->interrupt();
193 }
194 }
195 }
196
203 bool tryExecute(const _weak<CancellationWrapper<Inner>>& innerWeak) {
204 /*
205 * We should assume that this == nullptr or invalid.
206 * We can assert that this pointer is safe only if we hold at least one shared_ptr.
207 * It allows callers to pass a weak_ptr in any state.
208 */
209 if (auto innerCancellation = innerWeak.lock()) {
210 if (value) return false;
211 auto& inner = innerCancellation->ptr();
212 if (inner->setThread(AThread::current())) return false;
213 try {
214 if (task == nullptr) { // task is executed in wait() function
215 return false;
216 }
217 std::unique_lock lock(mutex);
218 if (task == nullptr) { // task is executed in wait() function
219 return false;
220 }
221 auto func = std::move(task);
222 assert(bool(func));
223 lock.unlock();
224 innerCancellation = nullptr;
225 if constexpr(isVoid) {
226 func();
227 if (auto sharedPtrLock = innerWeak.lock()) {
228 lock.lock();
229 value = true;
230 cv.notify_all();
231 notifyOnSuccessCallback();
232
233 (void)sharedPtrLock; // sharedPtrLock is *used*
234 lock.unlock(); // unlock earlier because destruction of shared_ptr may cause deadlock
235 }
236 } else {
237 auto result = func();
238 if (auto sharedPtrLock = innerWeak.lock()) {
239 lock.lock();
240 value = std::move(result);
241 cv.notify_all();
242 notifyOnSuccessCallback();
243
244 (void)sharedPtrLock; // sharedPtrLock is *used*
245 lock.unlock(); // unlock earlier because destruction of shared_ptr may cause deadlock
246 }
247 }
248 } catch (const AException&) {
249 if (auto sharedPtrLock = innerWeak.lock()) {
250 inner->reportException();
251 }
252 return false;
253 } catch (...) {
254 if (auto sharedPtrLock = innerWeak.lock()) {
255 inner->reportInterrupted();
256 }
257 throw;
258 }
259 }
260 return true;
261 }
262
263 void reportInterrupted() noexcept {
264 std::unique_lock lock(mutex);
265 interrupted = true;
266 cv.notify_all();
267 }
268
269 void reportException() noexcept {
270 std::unique_lock lock(mutex);
271 exception.emplace();
272 cv.notify_all();
273 AUI_NULLSAFE(onError)(*exception);
274 }
275
276
277 void notifyOnSuccessCallback() {
278 if (value && onSuccess) {
279 if constexpr(isVoid) {
280 onSuccess();
281 } else {
282 onSuccess(*value);
283 }
284 onSuccess = nullptr;
285 }
286 }
287 };

333 template<typename Callback>
334 void onSuccess(Callback&& callback) const noexcept {
335 std::unique_lock lock((*mInner)->mutex);
336 if constexpr(isVoid) {
337 if ((*mInner)->onSuccess) {
338 (*mInner)->onSuccess = [prev = std::move((*mInner)->onSuccess),
339 callback = std::forward<Callback>(callback)]() mutable {
340 prev();
341 callback();
342 };
343 } else {
344 (*mInner)->onSuccess = [callback = std::forward<Callback>(callback)]() mutable {
345 callback();
346 };
347 }
348 } else {
349 if ((*mInner)->onSuccess) {
350 (*mInner)->onSuccess = [prev = std::move((*mInner)->onSuccess),
351 callback = std::forward<Callback>(callback)](const Value& v) mutable {
352 prev(v);
353 callback(v);
354 };
355 } else {
356 (*mInner)->onSuccess = [callback = std::forward<Callback>(callback)](
357 const Value& v) mutable {
358 callback(v);
359 };
360 }
361 }
362 (*mInner)->notifyOnSuccessCallback();
363 }
364
365 template<aui::invocable<const AException&> Callback>
366 void onError(Callback&& callback) const noexcept {
367 std::unique_lock lock((*mInner)->mutex);
368
369 if ((*mInner)->onError) {
370 (*mInner)->onError = [prev = std::move((*mInner)->onError),
371 callback = std::forward<Callback>(callback)](const AException& v) {
372 prev(v);
373 callback(v);
374 };
375 } else {
376 (*mInner)->onError = [callback = std::forward<Callback>(callback)](
377 const AException& v) {
378 callback(v);
379 };
380 }
381 }
382
383
393 void cancel() noexcept {
394 (*mInner)->cancel();
395 }
396
397 void reportInterrupted() {
398 (*mInner)->reportInterrupted();
399 }
400
406 void wait(AFutureWait flags = AFutureWait::DEFAULT) noexcept {
407 (*mInner)->wait(mInner, flags);
408 }
409
418 typename FutureReturnType<Value>::type get(AFutureWait flags = AFutureWait::DEFAULT) {
419 AThread::interruptionPoint();
420 (*mInner)->wait(mInner, flags);
421 AThread::interruptionPoint();
422 if ((*mInner)->exception) {
423 throw *(*mInner)->exception;
424 }
425 if ((*mInner)->interrupted) {
426 throw AInvocationTargetException("Future execution interrupted");
427 }
428 if constexpr(!isVoid) {
429 return *(*mInner)->value;
430 }
431 }
432
433
442 typename FutureReturnType<Value>::type operator*() const {
443 return **const_cast<Future*>(this);
444 }
445
454 typename FutureReturnType<Value>::type operator*() {
455 return get();
456 }
457
466 Value* operator->() {
467 return &operator*();
468 }
469
478 Value const * operator->() const {
479 return &operator*();
480 }
481 };

527template<typename T = void>
528class AFuture final: public aui::impl::future::Future<T> {
529private:
530 using super = typename aui::impl::future::Future<T>;
531
532public:
533 using Task = typename super::TaskCallback;
534
535 AFuture(Task task = nullptr) noexcept: super(std::move(task)) {}
536 ~AFuture() = default;
537
544 void supplyResult(T v) const noexcept {
545 auto& inner = (*super::mInner);
546 assert(("task is already provided", inner->task == nullptr));
547
548 std::unique_lock lock(inner->mutex);
549 inner->value = std::move(v);
550 inner->cv.notify_all();
551 AUI_NULLSAFE(inner->onSuccess)(*inner->value);
552 }
553
557 void supplyException() const noexcept {
558 auto& inner = (*super::mInner);
559 inner->reportException();
560 }
561
562 AFuture& operator=(std::nullptr_t) noexcept {
563 super::mInner = nullptr;
564 return *this;
565 }
566
567 [[nodiscard]]
568 bool operator==(const AFuture& r) const noexcept {
569 return super::mInner == r.mInner;
570 }
571
591 template<aui::invocable<const T&> Callback>
592 const AFuture& onSuccess(Callback&& callback) const noexcept {
593 super::onSuccess(std::forward<Callback>(callback));
594 return *this;
595 }
596
616 template<aui::invocable<const AException&> Callback>
617 const AFuture& onError(Callback&& callback) const noexcept {
618 super::onError(std::forward<Callback>(callback));
619 return *this;
620 }
621
625 template<aui::invocable<const T&> Callback>
626 auto map(Callback&& callback) -> AFuture<decltype(callback(std::declval<T>()))> const {
627 AFuture<decltype(callback(std::declval<T>()))> result;
628 onSuccess([result, callback = std::forward<Callback>(callback)](const T& v) {
629 result.supplyResult(callback(v));
630 });
631 onError([result](const AException& v) {
632 try {
633 throw v;
634 } catch (...) {
635 result.reportException();
636 }
637 });
638 return result;
639 }
640};
