113 public:
114 static constexpr bool isVoid = std::is_same_v<void, Value>;
115 using TaskCallback = std::function<Value()>;

119 struct Inner {
120 bool interrupted = false;
124 std::conditional_t<isVoid, bool, AOptional<Value>> value;
125 AOptional<AInvocationTargetException> exception;
126 AMutex mutex;
127 AConditionVariable cv;
128 TaskCallback task;
129 OnSuccessCallback onSuccess;
130 std::function<void(const AException& exception)> onError;
131 _<AAbstractThread> thread;
132 bool cancelled = false;
133
134 explicit Inner(std::function<Value()> task) noexcept: task(std::move(task)) {
135 if constexpr(isVoid) {
136 value = false;
137 }
138 }
139
140 void waitForTask() noexcept {
141 std::unique_lock lock(mutex);
142 bool rethrowInterrupted = false;
143 while ((thread) && !hasResult() && !cancelled) {
144 try {
145 cv.wait(lock);
146 } catch (const AThread::Interrupted&) {
147 rethrowInterrupted = true;
148 }
149 }
150 if (rethrowInterrupted) {
151 AThread::current()->interrupt();
152 }
153 }
154
155 [[nodiscard]]
156 bool isWaitNeeded() noexcept {
157 return (thread || !cancelled) && !hasResult();
158 }
159
160 [[nodiscard]]
161 bool hasResult() const noexcept {
162 return value || exception || interrupted;
163 }
164
165 [[nodiscard]]
166 bool hasValue() const noexcept {
167 return bool(value);
168 }
169
170 bool setThread(_<AAbstractThread> thr) noexcept {
171 std::unique_lock lock(mutex);
172 if (cancelled) return true;
173 if (thread) return true;
174 thread = std::move(thr);
175 return false;
176 }
177
178 void wait(const _weak<CancellationWrapper<Inner>>& innerWeak, ABitField<AFutureWait> flags = AFutureWait::DEFAULT) noexcept;
179
180 void cancel() noexcept {
181 std::unique_lock lock(mutex);
182 if (!cancelled) {
183 cancelled = true;
184 if (thread && !hasResult()) {
185 thread->interrupt();
186 }
187 }
188 }
189
196 bool tryExecute(const _weak<CancellationWrapper<Inner>>& innerWeak) {
197 /*
198 * We should assume that this == nullptr or invalid.
199 * We can assert that this pointer is safe only if we hold at least one shared_ptr.
200 * It allows callers to pass a weak_ptr in any state.
201 */
202 if (auto innerCancellation = innerWeak.lock()) {
203 if (value) return false;
204 auto& inner = innerCancellation->ptr();
205 if (inner->setThread(AThread::current())) return false;
206 try {
207 if (task == nullptr) { // task is executed in wait() function
208 return false;
209 }
210 std::unique_lock lock(mutex);
211 if (task == nullptr) { // task is executed in wait() function
212 return false;
213 }
214 auto func = std::move(task);
215 AUI_ASSERT(bool(func));
216 lock.unlock();
217 innerCancellation = nullptr;
218 if constexpr(isVoid) {
219 func();
220 if (auto sharedPtrLock = innerWeak.lock()) {
221 lock.lock();
222 value = true;
223 cv.notify_all();
224 notifyOnSuccessCallback(lock);
225
226 (void)sharedPtrLock; // sharedPtrLock is *used*
227 if (lock.owns_lock()) lock.unlock(); // unlock earlier because destruction of shared_ptr may cause deadlock
228 }
229 } else {
230 auto result = func();
231 if (auto sharedPtrLock = innerWeak.lock()) {
232 lock.lock();
233 value = std::move(result);
234 cv.notify_all();
235 notifyOnSuccessCallback(lock);
236
237 (void)sharedPtrLock; // sharedPtrLock is *used*
238 if (lock.owns_lock()) lock.unlock(); // unlock earlier because destruction of shared_ptr may cause deadlock
239 }
240 }
241 } catch (const AThread::Interrupted&) {
242 if (auto sharedPtrLock = innerWeak.lock()) {
243 inner->reportInterrupted();
244 }
245 throw;
246 } catch (...) {
247 if (auto sharedPtrLock = innerWeak.lock()) {
248 inner->reportException();
249 }
250 return false;
251 }
252 }
253 return true;
254 }
255
256 void reportInterrupted() noexcept {
257 std::unique_lock lock(mutex);
258 interrupted = true;
259 cv.notify_all();
260 }
261
262 void reportException() noexcept {
263 if (cancelled) {
264 return;
265 }
266 std::unique_lock lock(mutex);
267 exception.emplace();
268 cv.notify_all();
269 if (!onError) {
270 return;
271 }
272 auto localOnError = std::move(onError);
273 lock.unlock();
274 localOnError(*exception);
275 }
276
277
287 void notifyOnSuccessCallback(std::unique_lock<decltype(mutex)>& lock) noexcept {
288 AUI_ASSERT(lock.owns_lock());
289 if (cancelled) {
290 return;
291 }
292 if (value && onSuccess) {
293 auto localOnSuccess = std::move(onSuccess);
294 onSuccess = nullptr;
295 lock.unlock();
296 invokeOnSuccessCallback(localOnSuccess);
297 }
298 }

300 template<typename F>
301 void invokeOnSuccessCallback(F&& f) {
302 try {
303 if constexpr (isVoid) {
304 f();
305 } else {
306 f(*value);
307 }
308 } catch (const AException& e) {
309 ALogger::err("AFuture") << "AFuture onSuccess thrown an exception: " << e;
310 }
311 }
312
313 template<typename Callback>
314 void addOnSuccessCallback(Callback&& callback) {
315 if constexpr (isVoid) {
316 if (onSuccess) {
317 onSuccess = [prev = std::move(onSuccess),
318 callback = std::forward<Callback>(callback)]() mutable {
319 prev();
320 callback();
321 };
322 } else {
323 onSuccess = [callback = std::forward<Callback>(callback)]() mutable { callback(); };
324 }
325 } else {
326 if (onSuccess) {
327 onSuccess = [prev = std::move(onSuccess),
328 callback = std::forward<Callback>(callback)](const Value& v) mutable {
329 prev(v);
330 callback(v);
331 };
332 } else {
333 onSuccess = [callback = std::forward<Callback>(callback)](const Value& v) mutable {
334 callback(v);
335 };
336 }
337 }
338 }
339
340 template<typename Callback>
341 void addOnErrorCallback(Callback&& callback) {
342 if (onError) {
343 onError = [prev = std::move(onError),
344 callback = std::forward<Callback>(callback)](const AException& v) {
345 prev(v);
346 callback(v);
347 };
348 } else {
349 onError = [callback = std::forward<Callback>(callback)](const AException& v) { callback(v); };
350 }
351 }
352 };

405 template<typename Callback>
406 void onSuccess(Callback&& callback) const noexcept {
407 if (hasValue()) { // cheap lookahead
408 (*mInner)->invokeOnSuccessCallback(std::forward<Callback>(callback));
409 return;
410 }
411 std::unique_lock lock((*mInner)->mutex);
412 if (hasValue()) { // not so cheap, but cheapier than adding the callback to inner
413 (*mInner)->invokeOnSuccessCallback(std::forward<Callback>(callback));
414 return;
415 }
416 (*mInner)->addOnSuccessCallback(std::forward<Callback>(callback));
417 }

421 std::unique_lock lock((*mInner)->mutex);
422 (*mInner)->addOnErrorCallback(std::forward<Callback>(callback));
423 }

460 typename FutureReturnType<Value>::type get(AFutureWait flags = AFutureWait::DEFAULT) const {
461 AThread::interruptionPoint();
462 (*mInner)->wait(mInner, flags);
463 AThread::interruptionPoint();
464 if ((*mInner)->exception) {
465 throw *(*mInner)->exception;

472 }
473 }

597template<typename T = void>
598class AFuture final: public aui::impl::future::Future<T> {
599private:
600 using super = typename aui::impl::future::Future<T>;
601
602public:
603 using Task = typename super::TaskCallback;
604#if AUI_COROUTINES
605 using promise_type = typename super::CoPromiseType;
606#endif
607 using Inner = aui::impl::future::CancellationWrapper<typename super::Inner>;
608
609 explicit AFuture(T immediateValue): super() {
610 auto& inner = (*super::mInner);
611 inner->value = std::move(immediateValue);
612 }
613 AFuture(Task task = nullptr) noexcept: super(std::move(task)) {}
614 ~AFuture() = default;
615
622 void supplyResult(T v) const noexcept {
623 auto& inner = (*super::mInner);
624 AUI_ASSERTX(inner->task == nullptr, "task is already provided");
625
626 std::unique_lock lock(inner->mutex);
627 inner->value = std::move(v);
628 inner->cv.notify_all();
629 inner->notifyOnSuccessCallback(lock);
630 }
631
635 void supplyException() const noexcept {
636 auto& inner = (*super::mInner);
637 inner->reportException();
638 }
639
640 AFuture& operator=(std::nullptr_t) noexcept {
641 super::mInner = nullptr;
642 return *this;
643 }
644
645 [[nodiscard]]
646 bool operator==(const AFuture& r) const noexcept {
647 return super::mInner == r.mInner;
648 }
649
669 template<aui::invocable<const T&> Callback>
670 const AFuture& onSuccess(Callback&& callback) const noexcept {
671 super::onSuccess(std::forward<Callback>(callback));
672 return *this;
673 }
674
694 template<aui::invocable<const AException&> Callback>
695 const AFuture& onError(Callback&& callback) const noexcept {
696 super::onError(std::forward<Callback>(callback));
697 return *this;
698 }
699
703 template<aui::invocable<const T&> Callback>
704 auto map(Callback&& callback) -> AFuture<decltype(callback(std::declval<T>()))> const {
705 AFuture<decltype(callback(std::declval<T>()))> result;
706 onSuccess([result, callback = std::forward<Callback>(callback)](const T& v) {
707 result.supplyResult(callback(v));
708 });
709 onError([result](const AException& v) {
710 try {
711 throw v;
712 } catch (...) {
713 result.reportException();
714 }
715 });
716 return result;
717 }
718};

824template <typename Value>
825void aui::impl::future::Future<Value>::Inner::wait(const _weak<CancellationWrapper<Inner>>& innerWeak,
826 ABitField<AFutureWait> flags) noexcept {
827 if (hasResult()) return; // cheap check
828 std::unique_lock lock(mutex);
829 try {
830 if ((thread || !cancelled) && !hasResult() && flags & AFutureWait::ALLOW_TASK_EXECUTION_IF_NOT_PICKED_UP && task) {
831 // task is not have picked up by the threadpool; execute it here
832 lock.unlock();
833 if (tryExecute(innerWeak)) {
834 return;
835 }
836 lock.lock();
837 }
838
839 if (flags & AFutureWait::ALLOW_STACKFUL_COROUTINES) {
840 if (auto threadPoolWorker = _cast<AThreadPool::Worker>(AThread::current())) {
841 if (hasResult()) return; // for sure
842 AUI_ASSERT(lock.owns_lock());
843 auto callback = [threadPoolWorker](auto&&...) {
844 threadPoolWorker->threadPool().wakeUpAll();
845 };
846 addOnSuccessCallback(callback);
847 addOnErrorCallback(callback);
848
849 threadPoolWorker->loop([&] {
850 if (lock.owns_lock())
851 lock.unlock();
852 return !hasResult();
853 });
854
855 return;
856 }
857 }
858 while ((thread || !cancelled) && !hasResult()) {
859 cv.wait(lock);
860 }
861 } catch (const AThread::Interrupted& e) {
862 e.needRethrow();
863 }
864}
