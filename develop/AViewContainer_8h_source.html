118 template<aui::mapper<const _<AView>&, bool> Callback>
119 bool getViewAtRecursive(glm::ivec2 pos, const Callback& callback, bool ignoreGone = true) {
120 for (auto it = mViews.rbegin(); it != mViews.rend(); ++it) {
121 auto view = *it;
122 auto targetPos = pos - view->getPosition();
123
124 if (targetPos.x >= 0 && targetPos.y >= 0 && targetPos.x < view->getSize().x &&
125 targetPos.y < view->getSize().y) {
126 if (!ignoreGone || view->getVisibility() != Visibility::GONE) {
127 if (callback(view))
128 return true;
129 if (auto container = _cast<AViewContainer>(view)) {
130 if (container->getViewAtRecursive(targetPos, callback, ignoreGone)) {
131 return true;
132 }
133 }
134 }
135 }
136 }
137 return false;
138 }

140 template<aui::mapper<const _<AView>&, bool> Callback>
141 bool getViewRecursive(const Callback& callback, bool ignoreGone = true) {
142 for (auto it = mViews.rbegin(); it != mViews.rend(); ++it) {
143 auto view = *it;
144 if (!ignoreGone || view->getVisibility() != Visibility::GONE) {
145 if (callback(view))
146 return true;
147 if (auto container = _cast<AViewContainer>(view)) {
148 if (container->getViewRecursive(callback, ignoreGone)) {
149 return true;
150 }
151 }
152 }
153 }
154 return false;
155 }
