111 template<typename T>
112 _<T> getViewAtRecursiveOf(glm::ivec2 pos, bool ignoreGone = true) {
113 for (auto it = mViews.rbegin(); it != mViews.rend(); ++it)
114 {
115 auto view = *it;
116 auto targetPos = pos - view->getPosition();
117
118 if (targetPos.x >= 0 && targetPos.y >= 0 && targetPos.x < view->getSize().x && targetPos.y < view->getSize().y)
119 {
120 if (!ignoreGone || view->getVisibility() != Visibility::GONE) {
121 if (auto applicable = _cast<T>(view))
122 return applicable;
123 if (auto container = _cast<AViewContainer>(view)) {
124 if (auto applicable = container->getViewAtRecursiveOf<T>(targetPos, ignoreGone)) {
125 return applicable;
126 }
127 }
128 }
129 }
130 }
131 return nullptr;
132 }
