99 if (auto index = std::get_if<ATreeModelIndex>(&vertex)) {
100 if (p(*index)) {
101 return *index;

103 }
104 std::size_t count = childrenCount(vertex);
105 for (std::size_t i = 0; i < count; ++i) {

109 }
