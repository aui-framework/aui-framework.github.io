161 template<typename OtherIterator>
162 constexpr iterator insert(iterator at, OtherIterator begin, OtherIterator end) {
163 AUI_ASSERT_MY_ITERATOR(at);
164 auto distance = std::distance(begin, end);
165
166 if (distance + size() <= StaticVectorSize) {
167 return inplace()->insert(at, begin, end);
168 }
169 if (!isInplaceAllocated()) {
170 return dynamic()->insert(at, begin, end);
171 }
172
173 // switch from static to dynamic
174 DynamicVector temp;
175 temp.reserve(aui::container::vector_impl::ceilPower2(distance + size()));
176
177 aui::container::vector_impl::insert_no_growth(temp.mEnd, temp.mEnd,
178 std::make_move_iterator(this->begin()), std::make_move_iterator(at));
179
180 auto result = aui::container::vector_impl::insert_no_growth(temp.mEnd, temp.mEnd,
181 begin, end);
182
183 aui::container::vector_impl::insert_no_growth(temp.mEnd, temp.mEnd,
184 std::make_move_iterator(at), std::make_move_iterator(this->end()));
185
186 inplace()->~StaticVector();
187 new (&mBase.dynamic) DynamicVector(std::move(temp));
188
189 return result;
190 }
191
192 constexpr iterator erase(iterator begin, iterator end) noexcept {
193 AUI_ASSERT_MY_ITERATOR(begin);
194 AUI_ASSERT_MY_ITERATOR(end);
195
196 if (isInplaceAllocated()) {
197 return inplace()->erase(begin, end);
198 }
199
200 if (size() - std::distance(begin, end) > StaticVectorSize) {
201 return dynamic()->erase(begin, end);
202 }
203
204 auto index = std::distance(this->begin(), begin);
205
206 // switch from dynamic to static
207 auto temp = std::move(*dynamic());
208 temp.erase(begin, end);
209 new (inplace()) StaticVector();
210 inplace()->insert(inplace()->end(), std::make_move_iterator(temp.begin()), std::make_move_iterator(temp.end()));
211 return this->begin() + index;
212 }
