223 _<AOverlappingSurface> createOverlappingSurface(const glm::ivec2& position,
224 const glm::ivec2& size,
225 bool closeOnClick = true) {
226 return createOverlappingSurface([&](unsigned attempt) -> AOptional<glm::ivec2> {
227 switch (attempt) {
228 case 0: return position;
229 case 1: return glm::clamp(position, {0, 0}, {getSize() - size});
230 default: return std::nullopt;
231 }
232 }, size, closeOnClick);
233 }
234
245 _<AOverlappingSurface> createOverlappingSurface(const std::function<AOptional<glm::ivec2>(unsigned)>& positionFactory,
246 const glm::ivec2& size,
247 bool closeOnClick = true) {
248 glm::ivec2 position = {0, 0};
249 auto maxPos = getSize() - size;
250 for (unsigned index = 0; ; ++index) {
251 auto optionalPosition = positionFactory(index);
252 if (optionalPosition) {
253 position = *optionalPosition;
254
255 if (position.x >= 0 && position.y >= 0 && glm::all(glm::lessThan(position, maxPos))) {
256 break;
257 }
258 } else {
259 break;
260 }
261 }
262
263 auto tmp = createOverlappingSurfaceImpl(position, size);
264 tmp->mParentWindow = this;
265 tmp->mCloseOnClick = closeOnClick;
266 mOverlappingSurfaces << tmp;
267 return tmp;
268 }
269 void closeOverlappingSurface(AOverlappingSurface* surface) {
270 if (mOverlappingSurfaces.erase(aui::ptr::fake(surface)) > 0) {
271 closeOverlappingSurfaceImpl(surface);
272 }
273 }
