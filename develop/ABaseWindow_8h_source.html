177 _<AOverlappingSurface> createOverlappingSurface(const glm::ivec2& position,
178 const glm::ivec2& size,
179 bool closeOnClick = true) {
180 return createOverlappingSurface([&](unsigned attempt) -> AOptional<glm::ivec2> {
181 switch (attempt) {
182 case 0: return position;
183 case 1: return glm::clamp(position, {0, 0}, {getSize() - size});
184 default: return std::nullopt;
185 }
186 }, size, closeOnClick);
187 }

199 _<AOverlappingSurface> createOverlappingSurface(const std::function<AOptional<glm::ivec2>(unsigned)>& positionFactory,
200 const glm::ivec2& size,
201 bool closeOnClick = true) {
202 glm::ivec2 position = {0, 0};
203 auto maxPos = getSize() - size;
204 for (unsigned index = 0; ; ++index) {
205 auto optionalPosition = positionFactory(index);
206 if (optionalPosition) {
207 position = *optionalPosition;
208
209 if (position.x >= 0 && position.y >= 0 && glm::all(glm::lessThan(position, maxPos))) {
210 break;
211 }
212 } else {
213 break;
214 }
215 }
216
217 auto tmp = createOverlappingSurfaceImpl(position, size);
218 tmp->mParentWindow = this;
219 tmp->mCloseOnClick = closeOnClick;
220 mOverlappingSurfaces << tmp;
221 return tmp;
222 }
223 void closeOverlappingSurface(AOverlappingSurface* surface) {
224 if (mOverlappingSurfaces.erase(aui::ptr::fake(surface)) > 0) {
225 closeOverlappingSurfaceImpl(surface);
226 }
227 }
