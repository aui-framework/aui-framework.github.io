194 _<AOverlappingSurface> createOverlappingSurface(const glm::ivec2& position,
195 const glm::ivec2& size,
196 bool closeOnClick = true) {
197 return createOverlappingSurface([&](unsigned attempt) -> AOptional<glm::ivec2> {
198 switch (attempt) {
199 case 0: return position;
200 case 1: return glm::clamp(position, {0, 0}, {getSize() - size});
201 default: return std::nullopt;
202 }
203 }, size, closeOnClick);
204 }

216 _<AOverlappingSurface> createOverlappingSurface(const std::function<AOptional<glm::ivec2>(unsigned)>& positionFactory,
217 const glm::ivec2& size,
218 bool closeOnClick = true) {
219 glm::ivec2 position = {0, 0};
220 auto maxPos = getSize() - size;
221 for (unsigned index = 0; ; ++index) {
222 auto optionalPosition = positionFactory(index);
223 if (optionalPosition) {
224 position = *optionalPosition;
225
226 if (position.x >= 0 && position.y >= 0 && glm::all(glm::lessThan(position, maxPos))) {
227 break;
228 }
229 } else {
230 break;
231 }
232 }
233
234 auto tmp = createOverlappingSurfaceImpl(position, size);
235 tmp->mParentWindow = this;
236 tmp->mCloseOnClick = closeOnClick;
237 mOverlappingSurfaces << tmp;
238 return tmp;
239 }
240 void closeOverlappingSurface(AOverlappingSurface* surface) {
241 if (mOverlappingSurfaces.erase(aui::ptr::fake(surface)) > 0) {
242 closeOverlappingSurfaceImpl(surface);
243 }
244 }
