166 _<AOverlappingSurface> createOverlappingSurface(const std::function<AOptional<glm::ivec2>(unsigned)>& positionFactory,
167 const glm::ivec2& size,
168 bool closeOnClick = true) {
169 glm::ivec2 position = {0, 0};
170 auto maxPos = getSize() - size;
171 for (unsigned index = 0; ; ++index) {
172 auto optionalPosition = positionFactory(index);
173 if (optionalPosition) {
174 position = *optionalPosition;
175
176 if (position.x >= 0 && position.y >= 0 && glm::all(glm::lessThan(position, maxPos))) {
177 break;
178 }
179 } else {
180 break;
181 }
182 }
183
184 auto tmp = createOverlappingSurfaceImpl(position, size);
185 tmp->mParentWindow = this;
186 tmp->mCloseOnClick = closeOnClick;
187 mOverlappingSurfaces << tmp;
188 return tmp;
189 }
190 void closeOverlappingSurface(AOverlappingSurface* surface) {
191 if (mOverlappingSurfaces.erase(aui::ptr::fake(surface)) > 0) {
192 closeOverlappingSurfaceImpl(surface);
193 }
194 }
