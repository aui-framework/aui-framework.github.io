150 _<AOverlappingSurface> createOverlappingSurface(const std::function<std::optional<glm::ivec2>(unsigned)>& positionFactory,
151 const glm::ivec2& size,
152 bool closeOnClick = true) {
153 glm::ivec2 position = {0, 0};
154 auto maxPos = getSize() - size;
155 for (unsigned index = 0; ; ++index) {
156 auto optionalPosition = positionFactory(index);
157 if (optionalPosition.has_value()) {
158 position = *optionalPosition;
159
160 if (position.x >= 0 && position.y >= 0 && glm::all(glm::lessThan(position, maxPos))) {
161 break;
162 }
163 } else {
164 break;
165 }
166 }
167
168 auto tmp = createOverlappingSurfaceImpl(position, size);
169 tmp->mWindow = this;
170 tmp->mCloseOnClick = closeOnClick;
171 mOverlappingSurfaces << tmp;
172 return tmp;
173 }
174 void closeOverlappingSurface(AOverlappingSurface* surface) {
175 if (mOverlappingSurfaces.erase(aui::ptr::fake(surface)) > 0) {
176 closeOverlappingSurfaceImpl(surface);
177 }
178 }
