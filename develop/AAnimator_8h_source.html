40 class CubicBezier {
41 private:
42 glm::vec2 v1, v2;
43 public:
44 CubicBezier(const glm::vec2& v1, const glm::vec2& v2) : v1(v1), v2(v2) {}
45
46 float operator()(float t) {
47 float oneMinusX = 1 - t;
48 float oneMinusX2 = oneMinusX * oneMinusX;
49 float oneMinusX3 = oneMinusX2 * oneMinusX;
50 float x2 = t * t;
51 float x3 = x2 * t;
52
53 float result = 3 * v1.x * oneMinusX2 * t + 3 * v2.x * oneMinusX * x2 + 2 * x3 +
54 3 * v1.y * oneMinusX2 * t + 3 * v2.y * oneMinusX * x2 - t;
55 return result;
56 }
57 };
58
59 class Standard: public CubicBezier {
60 public:
61 Standard(): CubicBezier({ 0.4f, 0.0f }, { 0.2f, 1.f }) {}
62 };
