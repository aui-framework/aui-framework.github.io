   50        static_assert(aui::has_json_converter<T>, "this type does not implement AJsonConv<T> trait");
   51        return AJsonConv<T>::toJson(t);
   52    }
   53    template<typename T>
   54    inline T from_json(const AJson& v) {
   55        try {
   56            static_assert(aui::has_json_converter<T>, "this type does not implement AJsonConv<T> trait");
   57            T dst;
   58            AJsonConv<T>::fromJson(v, dst);
   59            return dst;
   60        } catch (...) {
   61            throw AJsonException("While converting from json to cpp\n" + AJson::toString(v), std::current_exception());
   62        }
   63    }

   93    template<typename T>
   94    struct Field {
   95        T& value;
   96        const char* name;
   97        AJsonFieldFlags flags;
   98
   99        Field(T& value, const char* name, AJsonFieldFlags flags) : value(value), name(name), flags(flags) {}
  100
  101        void operator()(const AJson::Object& object) {
  102            if (auto c = object.contains(name)) {
  103                aui::from_json<T>(c->second, value);
  104            } else {
  105                if (!(flags & AJsonFieldFlags::OPTIONAL)) {
  106                    throw AJsonException(R"(field "{}" is not present)"_format(name));
  107                }
  108            }
  109        }
  110        void operator()(AJson::Object& object) {
  111            object[name] = aui::to_json<T>(value);
  112        }
  113    };

  211template<typename T>
  212struct AJsonConv<T, std::enable_if_t<aui::is_complete<AJsonConvFieldDescriptor<T>>>> {
  213
  214    static AJson toJson(const T& t) {
  215        AJson::Object json;
  216        std::apply([&](auto&&... fields) {
  217            aui::parameter_pack::for_each([&](auto&& field) {
  218                field(json);
  219            }, fields...);
  220        }, ((AJsonConvFieldDescriptor<T>&)t)().stdTuple());
  221        return std::move(json);
  222    }
  223
  224    static void fromJson(const AJson& json, T& dst) {
  225        const auto& jsonObject = json.asObject();
  226        std::apply([&](auto&&... fields) {
  227            aui::parameter_pack::for_each([&](auto&& field) {
  228                field(jsonObject);
  229            }, fields...);
  230        }, ((AJsonConvFieldDescriptor<T>&)dst)().stdTuple());
  231    }
  232};

  359template<typename T>
  360struct AJsonConv<AVector<T>, typename std::enable_if_t<aui::has_json_converter<T>>> {
  361    static AJson toJson(const AVector<T>& v) {
  362        AJson::Array array;
  363        array.reserve(v.size());
  364        for (const auto& elem : v) {
  365            array << aui::to_json(elem);
  366        }
  367        return std::move(array);
  368    }
  369    static void fromJson(const AJson& json, AVector<T>& dst) {
  370        auto& array = json.asArray();
  371        dst.clear();
  372        dst.reserve(array.size());
  373        for (const auto& elem : array) {
  374            dst << aui::from_json<T>(elem);
  375        }
  376    }
