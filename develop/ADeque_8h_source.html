232template<typename T>
233inline std::ostream& operator<<(std::ostream& o, const ADeque<T>& v) {
234 if (v.empty()) {
235 o << "[empty]";
236 } else {
237 o << "[ " << v.first();
238 for (auto it = v.begin() + 1; it != v.end(); ++it) {
239 o << ", " << *it;
240 }
241 o << " ]";
242 }
