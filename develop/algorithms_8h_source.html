38 template<typename Predicate, typename Iterator>
39 struct BinarySearchNearestToZero {
40 public:
41 BinarySearchNearestToZero(Predicate predicate, Iterator end): mPredicate(std::move(predicate)), mEnd(end) {}
42
43 BinarySearchResult operator()(Iterator c) {
44 auto next = std::next(c);
45 if (next == mEnd) {
46 return BinarySearchResult::MATCH;
47 }
48
49 auto v1 = mPredicate(c);
50 auto v2 = mPredicate(next);
51
52 if (v1 <= 0 && v2 > 0) {
53 return BinarySearchResult::MATCH;
54 }
55 if (v1 > 0) {
56 return BinarySearchResult::LEFT;
57 }
58
59 return BinarySearchResult::RIGHT;
60 }
61 private:
62 Predicate mPredicate;
63 Iterator mEnd;
64 };

103 template<typename Iterator, typename Predicate>
104 Iterator binary_search(Iterator begin, Iterator end, Predicate predicate) {
105 static_assert(std::is_same_v<BinarySearchResult, decltype(predicate(std::declval<Iterator>()))>, "predicate should return BinarySearchResult");
106
107 size_t rangeEnd = end - begin;
108 size_t rangeBegin = 0;
109
110
111 for (size_t i; rangeBegin < rangeEnd; ) {
112 i = (rangeEnd - rangeBegin - 1) / 2 + rangeBegin;
113 auto it = begin + i;
114 switch (predicate(it)) {
115 case BinarySearchResult::MATCH:
116 return it;
117
118 case BinarySearchResult::RIGHT:
119 rangeBegin = i + 1;
120 break;
121 case BinarySearchResult::LEFT:
122 rangeEnd = i;
123 break;
124 }
125
126 }
127 return end;
128 }
