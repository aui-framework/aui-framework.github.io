  213        AColor accumulator;
  214        aui::zero(accumulator);
  215
  216        for (auto i = begin(); i != end(); ++i) {
  217            accumulator += AColor(*i);

  219
  220        return accumulator / float(width() * height());
  221    }

  224template <AImageViewVisitor Visitor>
  225auto AImageView::visit(Visitor&& visitor) const {
  226#define AUI_CASE(v)                                                                                           \
  227    case v: {                                                                                                 \
  228        switch (format() & APixelFormat::TYPE_BITS) {                                                         \
  229            case APixelFormat::BYTE:                                                                          \
  230                return visitor(reinterpret_cast<const AFormattedImageView<v | APixelFormat::BYTE>&>(*this));  \
  231            case APixelFormat::FLOAT:                                                                         \
  232                return visitor(reinterpret_cast<const AFormattedImageView<v | APixelFormat::FLOAT>&>(*this)); \
  233        }                                                                                                     \
  234    }
  235
  236    switch (format() & APixelFormat::COMPONENT_BITS) {
  237        AUI_CASE(APixelFormat::R)
  238        AUI_CASE(APixelFormat::RG)
  239        AUI_CASE(APixelFormat::RGB)
  240        AUI_CASE(APixelFormat::RGBA)
  241        AUI_CASE(APixelFormat::ARGB)
  242        AUI_CASE(APixelFormat::BGRA)
  243
  244        default:
  245            throw AException("unknown color format");
  246    }
  247#undef AUI_CASE
  248}
